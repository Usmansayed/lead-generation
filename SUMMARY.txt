   h2.parent            | <a href="/path">              | Article URL
   article h3           | <h3 class="...">              | Preview text
   a[href*="/@"]        | <a href="/@username">         | Author link

4. CRITICAL FINDING - URL EXTRACTION
   - URLs are NOT inside h2 tags
   - URLs are in h2's PARENT <a> tag
   - Access via: h2.parent.get("href")

from selenium import webdriver
from selenium.webdriver.chrome.options import Options
from bs4 import BeautifulSoup
import time

# Setup
options = Options()
options.add_argument("--headless=new")
driver = webdriver.Chrome(options=options)

# Load page
driver.get("https://medium.com/search?q=startup+hiring")
time.sleep(15)  # Wait for Cloudflare

# Parse
soup = BeautifulSoup(driver.page_source, "html.parser")
articles = soup.find_all("article")

# Extract
for article in articles:
    h2 = article.find("h2")
    if h2:
        title = h2.get_text(strip=True)
        
        # URL is in h2's parent <a> tag
        url = None
        if h2.parent and h2.parent.name == "a":
            url = h2.parent.get("href")
            if url and not url.startswith("http"):
                url = f"https://medium.com{url}"
        
        preview = article.find("h3").get_text(strip=True)
        author_link = article.find("a", href=lambda x: x and "/@" in x)
        author = author_link.get_text(strip=True) if author_link else None
        
        print(f"Title: {title}")
        print(f"URL: {url}")
        print(f"Author: {author}")

driver.quit()
